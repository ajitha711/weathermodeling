import math

with open("input.txt", "r") as file:
    for line in file:
        a, b, c = map(float, line.strip().split())
        print(f"\nSolving for: a={a}, b={b}, c={c}")
        discriminant = b**2 - 4*a*c
        if discriminant >= 0:
            x1 = (-b + math.sqrt(discriminant)) / (2*a)
            x2 = (-b - math.sqrt(discriminant)) / (2*a)
            print("  Roots:", x1, x2)
        else:
            print("  No real roots")
